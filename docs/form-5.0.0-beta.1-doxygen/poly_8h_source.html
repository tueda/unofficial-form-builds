192  * expanded.

194 inline void poly::check_memory (int i) {
195  POLY_GETIDENTITY(*this);
196  if (i + 3 + AN.poly_num_vars + AM.MaxTal >= size_of_terms) expand_memory(i + AM.MaxTal);
