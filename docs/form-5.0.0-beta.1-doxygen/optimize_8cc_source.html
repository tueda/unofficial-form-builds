285 vector<vector<WORD> > get_brackets () {
286 
287  // check for brackets in expression
288  bool has_brackets = false;
289  for (WORD *t=optimize_expr; *t!=0; t+=*t) {
290  WORD *tend=t+*t; tend-=ABS(*(tend-1));
291  for (WORD *t1=t+1; t1<tend; t1+=*(t1+1))
292  if (*t1 == HAAKJE)
293  has_brackets=true;
294  }
295 
296  // replace brackets by SEPARATESYMBOL
297  vector<vector<WORD> > brackets;
298 
299  if (has_brackets) {
300  int exprlen=10; // we need potential space for an empty bracket
301  for (WORD *t=optimize_expr; *t!=0; t+=*t)
302  exprlen += *t;
303  WORD *newexpr = (WORD *)Malloc1(exprlen*sizeof(WORD), "optimize newexpr");
304 
305  int i=0;
306  int sep_power = 0;
307 
308  for (WORD *t=optimize_expr; *t!=0; t+=*t) {
309  WORD *t1 = t+1;
310 
311  // scan for bracket
312  vector<WORD> bracket;
313  for (; *t1!=HAAKJE; t1+=*(t1+1))
314  bracket.insert(bracket.end(), t1, t1+*(t1+1));
315 
316  if (brackets.size()==0 || bracket!=brackets.back()) {
317  sep_power++;
318  brackets.push_back(bracket);
319  }
320  t1+=*(t1+1);
321 
322  WORD left = t + *t - t1;
323  bool more_symbols = (left != ABS(*(t+*t-1)));
324 
325  // add power of SEPARATESYMBOL
326  newexpr[i++] = 1 + left + (more_symbols ? 2 : 4);
327  newexpr[i++] = SYMBOL;
328  newexpr[i++] = (more_symbols ? *(t1+1) + 2 : 4);
329  newexpr[i++] = SEPARATESYMBOL;
330  newexpr[i++] = sep_power;
331 
332  // add remaining terms
333  if (more_symbols) {
334  t1+=2;
335  left-=2;
336  }
337  while (left-->0)
338  newexpr[i++] = *(t1++);
339  }
340 /*
341  We insert here an empty bracket that is zero.
342  It is used for the case that there is only a single bracket which is
343  outside the notation for trees at a later stage.
344 
345  There may be a problem now in that in the case of sep_power==1
346  newexpr is bigger than optimize_expr. We have to check that.
347 */
348  if ( sep_power == 1 )
349  {
350  WORD *t;
351  vector<WORD> bracket;
352  bracket.push_back(0);
353  bracket.push_back(0);
354  bracket.push_back(0);
355  bracket.push_back(0);
356  sep_power++;
357  brackets.push_back(bracket);
358  newexpr[i++] = 8;
359  newexpr[i++] = SYMBOL;
360  newexpr[i++] = 4;
361  newexpr[i++] = SEPARATESYMBOL;
362  newexpr[i++] = sep_power;
363  newexpr[i++] = 1;
364  newexpr[i++] = 1;
365  newexpr[i++] = 3;
366  newexpr[i++] = 0;
367  for (t=optimize_expr; *t!=0; t+=*t) {}
368  if ( t-optimize_expr+1 < i ) { // We have to redo this
369  M_free(optimize_expr,"$-sort space");
370  optimize_expr = (WORD *)Malloc1(i*sizeof(WORD),"$-sort space");
371  }
372  }
373  else {
374  newexpr[i++] = 0;
375  }
376  memcpy(optimize_expr, newexpr, i*sizeof(WORD));
377  M_free(newexpr,"optimize newexpr");
378 
379  // if factorized, replace SEP by FAC and remove brackets
380  if (brackets[0].size()>0 && brackets[0][2]==FACTORSYMBOL) {
381  for (WORD *t=optimize_expr; *t!=0; t+=*t) {
382  if (*t == ABS(*(t+*t-1))+1) continue;
383  if (t[1]==SYMBOL)
384  for (int i=3; i<t[2]; i+=2)
385  if (t[i]==SEPARATESYMBOL) t[i]=FACTORSYMBOL;
386  }
387  return vector<vector<WORD> >();
388  }
389  }
390 
391  return brackets;
392 }

405 int count_operators (const WORD *expr, bool print=false) {
406 
407  int n=0;
408  while (*(expr+n)!=0) n+=*(expr+n);
409 
410  int cntpow=0, cntmul=0, cntadd=0, sumpow=0;
411  WORD maxpowfac=1, maxpowsep=1;
412 
413  for (const WORD *t=expr; *t!=0; t+=*t) {
414  if (t!=expr) cntadd++; // new term
415  if (*t==ABS(*(t+*t-1))+1) continue; // only coefficient
416 
417  int cntsym=0;
418 
419  if (t[1]==SYMBOL)
420  for (int i=3; i<t[2]; i+=2) {
421  if (t[i]==FACTORSYMBOL) {
422  maxpowfac = max(maxpowfac, t[i+1]);
423  continue;
424  }
425  if (t[i]==SEPARATESYMBOL) {
426  maxpowsep = max(maxpowsep, t[i+1]);
427  continue;
428  }
429  if (t[i+1]>2) { // (extra)symbol power>2
430  cntpow++;
431  sumpow += (int)floor(log(t[i+1])/log(2.0)) + popcount(t[i+1]) - 1;
432  }
433  if (t[i+1]==2) cntmul++; // (extra)symbol squared
434  cntsym++;
435  }
436 
437  if (ABS(*(t+*t-1))!=3 || *(t+*t-2)!=1 || *(t+*t-3)!=1) cntsym++; // non +/-1 coefficient
438 
439  if (cntsym > 0) cntmul+=cntsym-1;
440  }
441 
442  cntadd -= maxpowfac-1;
443  cntmul += maxpowfac-1;
444 
445  cntadd -= maxpowsep-1;
446 
447  if (print)
448  MesPrint ("*** STATS: original %lP %lM %lA : %l", cntpow,cntmul,cntadd,sumpow+cntmul+cntadd);
449 
450  return sumpow+cntmul+cntadd;
451 }

502 vector<WORD> occurrence_order (const WORD *expr, bool rev) {
503 
504  // count the number of occurrences of variables
505  map<WORD,int> cnt;
506  for (const WORD *t=expr; *t!=0; t+=*t) {
507  if (*t == ABS(*(t+*t-1))+1) continue; // skip constant term
508  if (t[1] == SYMBOL)
509  for (int i=3; i<t[2]; i+=2)
510  cnt[t[i]]++;
511  }
512 
513  bool is_fac=false, is_sep=false;
514  if (cnt.count(FACTORSYMBOL)) {
515  is_fac=true;
516  cnt.erase(FACTORSYMBOL);
517  }
518  if (cnt.count(SEPARATESYMBOL)) {
519  is_sep=true;
520  cnt.erase(SEPARATESYMBOL);
521  }
522 
523  // determine the order of the variables
524  vector<pair<int,WORD> > cnt_order;
525  for (map<WORD,int>::iterator i=cnt.begin(); i!=cnt.end(); i++)
526  cnt_order.push_back(make_pair(i->second, i->first));
527  sort(cnt_order.rbegin(), cnt_order.rend());
528 
529  // create resulting order
530  vector<WORD> order;
531  for (int i=0; i<(int)cnt_order.size(); i++)
532  order.push_back(cnt_order[i].second);
533 
534  if (rev) reverse(order.begin(),order.end());
535 
536  // add FACTORSYMBOL/SEPARATESYMBOL
537  if (is_fac) order.insert(order.begin(), FACTORSYMBOL);
538  if (is_sep) order.insert(order.begin(), SEPARATESYMBOL);
539 
540  return order;
541 }

635 void build_Horner_tree (const WORD **terms, int numterms, int var, int maxvar, int pos, vector<WORD> *res) {
636 
637  GETIDENTITY;
638 
639  if (var == maxvar) {
640  // Horner tree is finished, so add remaining terms unfactorized
641  // (note: since only complete Horner schemes seem to be useful, numterms=1 here
642 
643  for (int fr=0; fr<numterms; fr++) {
644 
645  bool empty = true;
646  const WORD *t = terms[fr];
647 
648  // add symbols
649  if (*t != ABS(*(t+*t-1))+1)
650  for (int i=3+2*pos; i<t[2]; i+=2) {
651  res->push_back(SYMBOL);
652  res->push_back(4);
653  res->push_back(t[i]);
654  res->push_back(t[i+1]);
655  if (!empty) res->push_back(OPER_MUL);
656  empty = false;
657  }
658 
659  // if empty, add a 1, since the result should look like "... coeff *"
660  if (empty) {
661  res->push_back(SNUMBER);
662  res->push_back(5);
663  res->push_back(1);
664  res->push_back(1);
665  res->push_back(3);
666  }
667 
668  // add coefficient
669  res->push_back(SNUMBER);
670  WORD n = ABS(*(t+*t-1));
671  res->push_back(n+2);
672  for (int i=0; i<n; i++)
673  res->push_back(*(t+*t-n+i));
674  res->push_back(OPER_MUL);
675 
676  if (fr>0) res->push_back(OPER_ADD);
677  }
678 
679  // result should end with gcd of the terms; right now this never
680  // triggers, but if one would allow for incomplete Horner schemes,
681  // one should extract the gcd here
682  if (numterms > 1) {
683  res->push_back(SNUMBER);
684  res->push_back(5);
685  res->push_back(1);
686  res->push_back(1);
687  res->push_back(3);
688  res->push_back(OPER_MUL);
689  }
690  }
691  else {
692  // extract variable "var" and the gcd and proceed recursively
693 
694  WORD nnum = 0, nden = 0, ntmp = 0, ndum = 0;
695  UWORD *num = NumberMalloc("build_Horner_tree");
696  UWORD *den = NumberMalloc("build_Horner_tree");
697  UWORD *tmp = NumberMalloc("build_Horner_tree");
698  UWORD *dum = NumberMalloc("build_Horner_tree");
699 
700  // previous coefficient for gcd extraction or coefficient multiplication
701  int prev_coeff_idx = -1;
702 
703  for (int fr=0; fr<numterms;) {
704 
705  // find power of current term
706  WORD pow = getpower(terms[fr], var, pos);
707 
708  // find all terms with that power
709  int to=fr+1;
710  while (to<numterms && getpower(terms[to], var, pos) == pow) to++;
711 
712  // recursively build Horner tree of all terms proportional to var^pow
713  build_Horner_tree (terms+fr, to-fr, var+1, maxvar, pow==0?pos:pos+1, res);
714 
715  if (AN.poly_vars[var] != FACTORSYMBOL && AN.poly_vars[var] != SEPARATESYMBOL) {
716  // if normal symbol, find gcd(numerators) and gcd(denominators)
717  WORD n1 = res->at(res->size()-2) / 2;
718  WORD *t1 = &res->at(res->size()-2-2*ABS(n1));
719 
720  WORD *t2 = fr==0 ? t1 : &res->at(prev_coeff_idx);
721  WORD n2 = fr==0 ? n1 : *(t2+*(t2+1)-1) / 2;
722  if (fr>0) t2+=2;
723 
724  GcdLong_fix_args(BHEAD (UWORD *)t1,n1,(UWORD *)t2,n2,num,&nnum);
725  GcdLong_fix_args(BHEAD (UWORD *)t1+ABS(n1),ABS(n1),(UWORD *)t2+ABS(n2),ABS(n2),den,&nden);
726 
727  // divide out gcds; note: leading zeroes can be added here
728  for (int i=0; i<2; i++) {
729  if (i==1 && fr==0) break;
730 
731  WORD *t = i==0 ? t1 : t2;
732  WORD n = i==0 ? n1 : n2;
733 
734  DivLong_fix_args((UWORD *)t, n, num, nnum, tmp, &ntmp, dum, &ndum);
735  for (int j=0; j<ABS(ntmp); j++) *t++ = tmp[j];
736  for (int j=0; j<ABS(n)-ABS(ntmp); j++) *t++ = 0;
737 
738  if (SGN(ntmp) != SGN(n)) n=-n;
739 
740  DivLong_fix_args((UWORD *)t, n, den, nden, tmp, &ntmp, dum, &ndum);
741  for (int j=0; j<ABS(ntmp); j++) *t++ = tmp[j];
742  for (int j=0; j<ABS(n)-ABS(ntmp); j++) *t++ = 0;
743 
744  *t++ = SGN(n) * (2*ABS(n)+1);
745  }
746 
747  // add the addition operator
748  if (fr>0) res->push_back(OPER_ADD);
749 
750  // add the power of "var"
751  WORD nextpow = (to==numterms ? 0 : getpower(terms[to], var, pos));
752 
753  if (pow-nextpow > 0) {
754  res->push_back(SYMBOL);
755  res->push_back(4);
756  res->push_back(var);
757  res->push_back(pow-nextpow);
758  res->push_back(OPER_MUL);
759  }
760 
761  // add the extracted gcd
762  res->push_back(SNUMBER);
