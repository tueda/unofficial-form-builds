62 VOID Pack(UWORD *a, WORD *na, UWORD *b, WORD nb)
63 {
64  WORD c, sgn = 1, i;
65  UWORD *to,*from;
66  if ( (c = *na) == 0 ) {
67  MLOCK(ErrorMessageLock);
68  MesPrint("Caught a zero in Pack");
69  MUNLOCK(ErrorMessageLock);
70  return;
71  }
72  if ( nb == 0 ) {
73  MLOCK(ErrorMessageLock);
74  MesPrint("Division by zero in Pack");
75  MUNLOCK(ErrorMessageLock);
76  return;
77  }
78  if ( *na < 0 ) { sgn = -sgn; c = -c; }
79  if ( nb < 0 ) { sgn = -sgn; nb = -nb; }
80  *na = MaX(c,nb);
81  to = a + c;
82  i = *na - c;
83  while ( --i >= 0 ) *to++ = 0;
84  i = *na - nb;
85  from = b;
86  NCOPY(to,from,nb);
87  while ( --i >= 0 ) *to++ = 0;
88  if ( sgn < 0 ) *na = -*na;
89 }

101 {
102  UWORD *pos;

104  if ( na < 0 ) { na = -na; }

107  a += i;
108  a--;
109  pos = a + i;

111  while ( !(*pos) ) { na--; pos--; }
112  }
113  *denom = na;
114  if ( sgn < 0 ) i = -i;
115  *numer = i;
116 }
