35  File contains the routines for the tree structure of sparse tables
36  We insert elements by
37  InsTableTree(T,tp) with T the TABLES element and tp the pointer
38  to the indices.
39  We look for elements with
40  FindTableTree(T,tp,inc) with T the TABLES element, tp the pointer to the
41  indices or the function arguments and inc tells which of these options.
42  The tree is cleared with ClearTableTree(T) and we rebuild the tree
43  after a .store in which we lost a part of the table with
44  RedoTableTree(T,newsize)
45 
46  In T->tablepointers we have the lists of indices for each element.
47  Additionally for each element there is an extension. There are
48  TABLEEXTENSION WORDs reserved for that. The old system had two words
49  One for the element in the rhs of the compile buffer and one for
50  an additional rhs in case the original would be overwritten by a new
51  definition, but the old was fixed by .global and hence it should be possible
52  to restore it.
53  New use (new = 24-sep-2001)
54  rhs1,numCompBuffer1,rhs2,numCompBuffer2,usage
55  Hence TABLEEXTENSION will be 5. Note that for 64 bits the use of the
56  compiler buffer is overdoing it a bit, but it would be too complicated
57  to try to give it special code.
58 */

95  Searches for the element specified by the list of arguments.
96  If found, it returns -(the offset in T->tablepointers)

99  This number is always > 0, because we start from 1.

102 int InsTableTree(TABLES T, WORD *tp)
103 {
104  COMPTREE *boomlijst, *q, *p, *s;
105  WORD *v1, *v2, *v3, xstop;
106  int ip, iq, is;
107  if ( T->numtree + 1 >= T->MaxTreeSize ) {
108  if ( T->MaxTreeSize == 0 ) ClearTableTree(T);
109  else {
110  is = T->MaxTreeSize * 2;
111  s = (COMPTREE *)Malloc1(is*sizeof(COMPTREE),"InsTableTree");
112  for ( ip = 0; ip < T->MaxTreeSize; ip++ ) { s[ip] = T->boomlijst[ip]; }
113  if ( T->boomlijst ) M_free(T->boomlijst,"InsTableTree");
114  T->boomlijst = s;
115  T->MaxTreeSize = is;
116  }
117  }
118  boomlijst = T->boomlijst;
119  q = boomlijst + T->rootnum;
120  if ( q->right == -1 ) { /* First element */
121  T->numtree++;
122  s = boomlijst+T->numtree;
123  q->right = T->numtree;
124  s->parent = T->rootnum;
125  s->left = s->right = -1;
126  s->blnce = 0;
127  s->value = tp - T->tablepointers;
128  s->usage = 0;
129  return(T->numtree);
130  }
131  ip = q->right;
132  if ( T->numind >= 0 ) xstop = T->numind;
133  else xstop = *tp + 1;
134  while ( ip >= 0 ) {
135  p = boomlijst + ip;
136  v1 = T->tablepointers + p->value;
137  v2 = tp; v3 = tp + xstop;
138  while ( *v1 == *v2 && v2 < v3 ) { v1++; v2++; }
139  if ( v2 >= v3 ) return(-p->value);
140  if ( *v1 > *v2 ) {
141  iq = p->right;
142  if ( iq >= 0 ) { ip = iq; }
143  else {
144  T->numtree++;
145  is = T->numtree;
146  p->right = is;
147  s = boomlijst + is;
148  s->parent = ip; s->left = s->right = -1;
149  s->blnce = 0; s->value = tp - T->tablepointers;
150  s->usage = 0;
151  p->blnce++;
152  if ( p->blnce == 0 ) return(T->numtree);
153  goto balance;
154  }
155  }
156  else if ( *v1 < *v2 ) {
157  iq = p->left;
158  if ( iq >= 0 ) { ip = iq; }
159  else {
160  T->numtree++;
161  is = T->numtree;
162  s = boomlijst+is;
163  p->left = is;
164  s->parent = ip; s->left = s->right = -1;
165  s->blnce = 0; s->value = tp - T->tablepointers;
166  s->usage = 0;
167  p->blnce--;
168  if ( p->blnce == 0 ) return(T->numtree);
169  goto balance;
170  }
171  }
172  }
173  MesPrint("Serious problems in InsTableTree!\n");
174  Terminate(-1);
175  return(0);
176 balance:;
177  for (;;) {
178  p = boomlijst + ip;
179  iq = p->parent;
180  if ( iq == T->rootnum ) break;
181  q = boomlijst + iq;
182  if ( ip == q->left ) q->blnce--;
183  else q->blnce++;
184  if ( q->blnce == 0 ) break;
185  if ( q->blnce == -2 ) {
186  if ( p->blnce == -1 ) { /* single rotation */
187  q->left = p->right;
188  p->right = iq;
189  p->parent = q->parent;
190  q->parent = ip;
191  if ( boomlijst[p->parent].left == iq ) boomlijst[p->parent].left = ip;
192  else boomlijst[p->parent].right = ip;
193  if ( q->left >= 0 ) boomlijst[q->left].parent = iq;
194  q->blnce = p->blnce = 0;
195  }
196  else { /* double rotation */
197  s = boomlijst + is;
198  q->left = s->right;
199  p->right = s->left;
200  s->right = iq;
201  s->left = ip;
202  if ( p->right >= 0 ) boomlijst[p->right].parent = ip;
203  if ( q->left >= 0 ) boomlijst[q->left].parent = iq;
204  s->parent = q->parent;
205  q->parent = is;
206  p->parent = is;
207  if ( boomlijst[s->parent].left == iq )
208  boomlijst[s->parent].left = is;
209  else boomlijst[s->parent].right = is;
210  if ( s->blnce > 0 ) { q->blnce = s->blnce = 0; p->blnce = -1; }
211  else if ( s->blnce < 0 ) { p->blnce = s->blnce = 0; q->blnce = 1; }
212  else { p->blnce = s->blnce = q->blnce = 0; }
213  }
214  break;
215  }
216  else if ( q->blnce == 2 ) {
217  if ( p->blnce == 1 ) { /* single rotation */
218  q->right = p->left;
219  p->left = iq;
220  p->parent = q->parent;
221  q->parent = ip;
222  if ( boomlijst[p->parent].left == iq ) boomlijst[p->parent].left = ip;
223  else boomlijst[p->parent].right = ip;
224  if ( q->right >= 0 ) boomlijst[q->right].parent = iq;
225  q->blnce = p->blnce = 0;
226  }
227  else { /* double rotation */
228  s = boomlijst + is;
229  q->right = s->left;
230  p->left = s->right;
231  s->left = iq;
232  s->right = ip;
233  if ( p->left >= 0 ) boomlijst[p->left].parent = ip;
234  if ( q->right >= 0 ) boomlijst[q->right].parent = iq;
235  s->parent = q->parent;
236  q->parent = is;
237  p->parent = is;
238  if ( boomlijst[s->parent].left == iq ) boomlijst[s->parent].left = is;
239  else boomlijst[s->parent].right = is;
240  if ( s->blnce < 0 ) { q->blnce = s->blnce = 0; p->blnce = 1; }
241  else if ( s->blnce > 0 ) { p->blnce = s->blnce = 0; q->blnce = -1; }
242  else { p->blnce = s->blnce = q->blnce = 0; }
243  }
244  break;
245  }
246  is = ip; ip = iq;
247  }
248  return(T->numtree);
249 }

254 
255  To be used when a sparse table is trimmed due to a .store
256  We rebuild the tree. In the future one could try to become faster
257  at the cost of quite some complexity.
258  We need to keep the first 'size' elements in the boomlijst.

265 void RedoTableTree(TABLES T, int newsize)
266 {
267  WORD *tp;
268  int i;
269  ClearTableTree(T);
270  for ( i = 0, tp = T->tablepointers; i < newsize; i++ ) {
271  InsTableTree(T,tp);
272  tp += ABS(T->numind)+TABLEEXTENSION;
273  }
274 }

280  int FindTableTree(TABLES T,WORD *,arglist,int,inc)
281  Searches for the element specified by the list of arguments.
282  If found, it returns the offset in T->tablepointers
283  If not found, it will return -1
284  The list here is from the list of function arguments. Hence it

290 int FindTableTree(TABLES T, WORD *tp, int inc)
291 {
292  COMPTREE *boomlijst = T->boomlijst, *q = boomlijst + T->rootnum, *p;
293  WORD *v1, *v2, *v3, xstop;
294  int ip, iq;
295  if ( q->right == -1 ) return(-1);
296  ip = q->right;
297  if ( inc > 1 ) tp += inc-1;
298  if ( T->numind >= 0 ) xstop = T->numind;
299  else { /* We have to read the number of arguments first */
300  if ( *tp <= 0 ) return(-1); /* Cannot be! */
301  xstop = *tp+1;
302  }
303  while ( ip >= 0 ) {
304  p = boomlijst + ip;
305  v1 = T->tablepointers + p->value;
306  v2 = tp; v3 = v1 + xstop;
307  while ( *v1 == *v2 && v1 < v3 ) { v1++; v2 += inc; }
308  if ( v1 == v3 ) {
309  p->usage++;
310  return(p->value);
311  }
312  if ( *v1 > *v2 ) {
313  iq = p->right;
314  if ( iq >= 0 ) { ip = iq; }
315  else return(-1);
316  }
317  else if ( *v1 < *v2 ) {
318  iq = p->left;
319  if ( iq >= 0 ) { ip = iq; }
320  else return(-1);
321  }
322  }
323  MesPrint("Serious problems in FindTableTree\n");
324  Terminate(-1);
325  return(-1);
326 }
