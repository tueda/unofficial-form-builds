360 /*
361  * Macros to be inserted at the end of a structure to align the whole structure.
362  *
363  * In the currently available systems,
364  * sizeof(POSITION) >= sizeof(pointers) == sizeof(LONG) >= sizeof(int)
365  * >= sizeof(WORD) >= sizeof(UBYTE) = 1.
366  * (POSITION is defined in struct.h and contains only an off_t variable.)
367  * Thus, if we put members of a structure in this order and use those macros,
368  * then we can align the data without relying on extra paddings added by
369  * the compiler. For example,
370  * typedef struct {
371  * int *a;
372  * LONG b;
373  * WORD c[2];
374  * UBYTE d;
375  * PADPOINTER(1,0,2,1);
376  * } A;
377  * typedef struct {
378  * POSITION p;
379  * A a; // aligned same as pointers
380  * int *b;
381  * LONG c;
382  * UBYTE d;
383  * PADPOSITION(1,1,0,0,1+sizeof(A));
384  * } B;
385  * The cost for the use of those PADXXX macros is a padding (>= 1 byte) will
386  * be always inserted even in the case that no padding is actually needed.
387  *
388  * Numbers for the arguments have to be calculated manually and so very
389  * error-prone. Be careful!
390  *
391  * Note that there is a 32-bit system in which off_t is aligned on 8-byte
392  * boundary, (e.g., Cygwin with large file support), but still the above
393  * inequalities are satisfied.
394  *
395  * The legendary story of these macros--they fixed some problems in ancient
396  * times when compilers were unreliable and didn't know how to correctly compute
397  * structure paddings--has been handed down, though nowadays there are only
398  * disadvantages for them in practice (ancient compilers most likely can't
399  * compile C99 and C++98+TR1 sources anyway).
400  */
401 #define PADDUMMY(type, size) \
402  UBYTE d_u_m_m_y[alignof(type) - ((size) & (alignof(type) - 1))]
403 #define PADPOSITION(ptr_,long_,int_,word_,byte_) \
404  PADDUMMY(off_t, \
405  + sizeof(int *) * (ptr_) \
406  + sizeof(LONG) * (long_) \
407  + sizeof(int) * (int_) \
408  + sizeof(WORD) * (word_) \
409  + sizeof(UBYTE) * (byte_) \
410  )
411 #define PADPOINTER(long_,int_,word_,byte_) \
412  PADDUMMY(int *, \
413  + sizeof(LONG) * (long_) \
414  + sizeof(int) * (int_) \
415  + sizeof(WORD) * (word_) \
416  + sizeof(UBYTE) * (byte_) \
417  )
418 #define PADLONG(int_,word_,byte_) \
419  PADDUMMY(LONG, \
420  + sizeof(int) * (int_) \
421  + sizeof(WORD) * (word_) \
422  + sizeof(UBYTE) * (byte_) \
423  )
424 #define PADINT(word_,byte_) \
425  PADDUMMY(int, \
426  + sizeof(WORD) * (word_) \
427  + sizeof(UBYTE) * (byte_) \
428  )
429 #define PADWORD(byte_) \
430  PADDUMMY(WORD, \
431  + sizeof(UBYTE) * (byte_) \
432  )
