322 /*
323  #] inictable :
324  #[ findcommand :
325 
326  Checks whether a command is in the command table.
327  If so a pointer to the table element is returned.
328  If not we return 0.
329  Note that when a command is not in the table, we have
330  to test whether it is an id command without id. It should
331  then have the structure pattern = rhs. This should be done
332  in the calling routine.
333 */
334 
335 KEYWORD *findcommand(UBYTE *in)
336 {
337  int hi, med, lo, i;
338  UBYTE *s, c;
339  s = in;
340  while ( FG.cTable[*s] <= 1 ) s++;
341  if ( s > in && *s == '[' && s[1] == ']' ) s += 2;
342  if ( *s ) { c = *s; *s = 0; }
343  else c = 0;
344 /*
345  First do a binary search in the second table
346 */
347  lo = 0;
348  hi = sizeof(com2commands)/sizeof(KEYWORD)-1;
349  do {
350  med = ( hi + lo ) / 2;
351  i = StrICmp(in,(UBYTE *)com2commands[med].name);
352  if ( i == 0 ) { if ( c ) *s = c; return(com2commands+med); }
353  if ( i < 0 ) hi = med-1;
354  else lo = med+1;
355  } while ( hi >= lo );
356 /*
357  Now do a 'hashed' search in the first table. It is sequential.
358 */
359  i = tolower(*in) - 'a';
360  med = alfatable1[i];
361  hi = alfatable1[i+1];
362  while ( med < hi ) {
363  if ( StrICont(in,(UBYTE *)com1commands[med].name) == 0 )
364  { if ( c ) *s = c; return(com1commands+med); }
365  med++;
366  }
367  if ( c ) *s = c;
368 /*
369  Unrecognized. Too bad!
370 */
371  return(0);
372 }

712 int CompileSubExpressions(SBYTE *tokens)
713 {

717  int level, num, i, sumlevel = 0, sumtype = SYMTOSYM;
718  int retval, error = 0;
719 /*
720  Eliminate all subexpressions. They are marked by LPARENTHESIS,RPARENTHESIS
721 */
722  AC.CompileLevel++;
723  while ( *s != TENDOFIT ) {
724  if ( *s == TFUNOPEN ) {
725  if ( fill < s ) *fill = TENDOFIT;
726  t = fill - 1;
727  while ( t >= tokens && t[0] >= 0 ) t--;
728  if ( t >= tokens && *t == TFUNCTION ) {
729  t++; i = 0; while ( *t >= 0 ) i = 128*i + *t++;
730  if ( i == AM.sumnum || i == AM.sumpnum ) {
731  t = s + 1;
732  if ( *t == TSYMBOL || *t == TINDEX ) {
733  t++; i = 0; while ( *t >= 0 ) i = 128*i + *t++;
734  if ( s[1] == TINDEX ) {
735  i += AM.OffsetIndex;
736  sumtype = INDTOIND;
737  }
738  else sumtype = SYMTOSYM;
739  sumlevel = i;
740  }
741  }
742  }
743  *fill++ = *s++;
744  }
745  else if ( *s == TFUNCLOSE ) { sumlevel = 0; *fill++ = *s++; }
746  else if ( *s == LPARENTHESIS ) {
747 /*

749  If the subexpression is just an integer, whatever its length,
750  we should try to keep it.
751  This is important when we have a function with an integer
752  argument. In particular this is relevant for the MZV program.
753 */
754  t = s; level = 0;
755  while ( level >= 0 ) {
756  s++;
757  if ( *s == LPARENTHESIS ) level++;
758  else if ( *s == RPARENTHESIS ) level--;
759  else if ( *s == TENDOFIT ) {
760  MesPrint("&Unbalanced subexpression parentheses");
761  return(-1);
762  }
763  }
764  t++; *s = TENDOFIT;
765  if ( sumlevel > 0 ) { /* Inside sum. Add wildcard to prototype */
766  oldwork = w1 = AT.WorkPointer;
767  w2 = AC.ProtoType;

781  We had two parentheses and at least two bytes contents.

783  this minimum. Ignoring this might lead to really rare and
784  hard to find errors, years from now.

800  while ( --i >= 0 ) *fill++ = (SBYTE)(number[i]);
801  s++;
802  }
803  else if ( *s == TEMPTY ) s++;
804  else *fill++ = *s++;
805  }
806  *fill = TENDOFIT;
807 /*
808  At this stage there are no more subexpressions.
809  Hence we can do the basic compilation.
810 */
811  if ( AC.CompileLevel == 1 && AC.ToBeInFactors ) {
812  error = CodeFactors(tokens);
813  }
814  AC.CompileLevel--;

816  if ( error < 0 ) return(error);
817  return(retval);
818 }
